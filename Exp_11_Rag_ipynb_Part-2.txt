        "\n",

        "                with open(data_path, 'rb') as f:\n",
        "                    data = pickle.load(f)\n",
        "                info['chunks'] = len(data['chunks'])\n",
        "\n",
        "                for _, metadata in data['chunks']:\n",
        "                    if 'source' in metadata:\n",
        "                        info['sources'].add(metadata['source'])\n",
        "\n",
        "                info['created'] = time.ctime(os.path.getctime(data_path))\n",
        "            except:\n",
        "                pass\n",
        "\n",
        "        return info\n",
        "\n",
        "    def clear_cache(self, cache_name: str = None):\n",
        "        \"\"\"Clear the cache for a specific name or all caches\"\"\"\n",
        "        if cache_name:\n",
        "            index_path = os.path.join(self.config.cache_dir, f\"{cache_name}_index.faiss\")\n",
        "            data_path = os.path.join(self.config.cache_dir, f\"{cache_name}_data.pkl\")\n",
        "\n",
        "            if os.path.exists(index_path):\n",
        "                os.remove(index_path)\n",
        "            if os.path.exists(data_path):\n",
        "                os.remove(data_path)\n",
        "            print(f\"✅ Cleared cache for: {cache_name}\")\n",
        "        else:\n",
        "            if os.path.exists(self.config.cache_dir):\n",
        "                shutil.rmtree(self.config.cache_dir)\n",
        "                os.makedirs(self.config.cache_dir, exist_ok=True)\n",
        "                print(f\"✅ Cleared all caches in: {self.config.cache_dir}\")\n",
        "\n",
        "    def save_to_cache(self, name: str = \"default\") -> None:\n",
        "        \"\"\"Save processed data to cache\"\"\"\n",
        "        self.cache_name = name\n",
        "        print(f\"\\nSaving to cache: {name}\")\n",
        "\n",
        "        index_path = os.path.join(self.config.cache_dir, f\"{name}_index.faiss\")\n",
        "        faiss.write_index(self.index, index_path)\n",
        "\n",
        "        data = {\n",
        "            'chunks': [(chunk.page_content, chunk.metadata) for chunk in self.chunks],\n",
        "            'chunk_metadata': self.chunk_metadata,\n",
        "            'config': self.config\n",
        "        }\n",
        "\n",
        "        data_path = os.path.join(self.config.cache_dir, f\"{name}_data.pkl\")\n",
        "        with open(data_path, 'wb') as f:\n",
        "            pickle.dump(data, f)\n",
        "\n",
        "        print(f\"  ✓ Index saved to: {index_path}\")\n",
        "        print(f\"  ✓ Data saved to: {data_path}\")\n",
        "        print(f\"  ✓ Total chunks saved: {len(self.chunks)}\")\n",
        "\n",
        "    def load_from_cache(self, name: str = \"default\") -> bool:\n",
        "        \"\"\"Load processed data from cache\"\"\"\n",
        "        self.cache_name = name\n",
        "        index_path = os.path.join(self.config.cache_dir, f\"{name}_index.faiss\")\n",
        "        data_path = os.path.join(self.config.cache_dir, f\"{name}_data.pkl\")\n",
        "\n",
        "        if not (os.path.exists(index_path) and os.path.exists(data_path)):\n",
        "            return False\n",
        "\n",
        "        print(f\"\\nLoading from cache: {name}\")\n",
        "\n",
        "        self.index = faiss.read_index(index_path)\n",
        "\n",
        "        with open(data_path, 'rb') as f:\n",
        "            data = pickle.load(f)\n",
        "\n",
        "        from langchain_core.documents import Document\n",
        "        self.chunks = [\n",
        "            Document(page_content=content, metadata=metadata)\n",
        "            for content, metadata in data['chunks']\n",
        "        ]\n",
        "        self.chunk_metadata = data['chunk_metadata']\n",
        "\n",
        "        print(f\"  ✓ Loaded {len(self.chunks)} chunks\")\n",
        "        print(f\"  ✓ Index size: {self.index.ntotal} vectors\")\n",
        "\n",
        "        if self.chunks:\n",
        "            sources = set(m['source'] for m in self.chunk_metadata)\n",
        "            print(f\"  ✓ Sources: {', '.join(list(sources)[:3])}\")\n",
        "\n",
        "        return True\n",
        "\n",
        "    # ============= GENERATION MODEL METHODS =============\n",
        "\n",
        "    def load_generation_model(self):\n",
        "        \"\"\"Load the generation model\"\"\"\n",
        "        if self.generation_model_loaded:\n",
        "            return True\n",
        "\n",
        "        print(f\"\\n🔄 Loading generation model: {self.config.generation_model}\")\n",
        "        print(\"   (This may take a minute on first run)\")\n",
        "\n",
        "        try:\n",
        "            device = \"cuda:0\" if self.config.use_gpu and torch.cuda.is_available() else \"cpu\"\n",
        "\n",
        "            if \"t5\" in self.config.generation_model.lower():\n",
        "                self.tokenizer = AutoTokenizer.from_pretrained(self.config.generation_model)\n",
        "                self.model = AutoModelForSeq2SeqLM.from_pretrained(\n",
        "                    self.config.generation_model,\n",
        "                    torch_dtype=torch.float32\n",
        "                )\n",
        "\n",
        "                if device != \"cpu\":\n",
        "                    self.model = self.model.to(device)\n",
        "\n",
        "                self.model.eval()\n",
        "                self.generation_model_loaded = True\n",
        "                print(f\"  ✅ Generation model loaded successfully on {device}\")\n",
        "                return True\n",
        "\n",
        "            else:\n",
        "                self.generation_pipeline = pipeline(\n",
        "                    \"text2text-generation\",\n",
        "                    model=self.config.generation_model,\n",
        "                    device=0 if device != \"cpu\" else -1,\n",
        "                    max_length=self.config.max_new_tokens,\n",
        "                    temperature=self.config.temperature,\n",
        "                    do_sample=self.config.do_sample\n",
        "                )\n",
        "                self.generation_model_loaded = True\n",
        "                print(f\"  ✅ Generation pipeline loaded successfully\")\n",
        "                return True\n",
        "\n",
        "        except Exception as e:\n",
        "            print(f\"  ❌ Error loading model: {e}\")\n",
        "            print(\"  Falling back to template-based responses\")\n",
        "            self.generation_model_loaded = False\n",
        "            return False\n",
        "\n",
        "    # ============= SEARCH METHODS =============\n",
        "\n",
        "    def search(self, query: str, k: int = None) -> List[Dict]:\n",
        "        \"\"\"Search for relevant chunks\"\"\"\n",
        "        if self.index is None or self.index.ntotal == 0:\n",
        "            print(\"⚠️ No index available for search\")\n",
        "            return []\n",
        "\n",
        "        if k is None:\n",
        "            k = min(self.config.similarity_top_k, self.index.ntotal)\n",
        "\n",
        "        query_embedding = self.embedding_model.encode([query]).astype('float32')\n",
        "        distances, indices = self.index.search(query_embedding, k)\n",
        "\n",
        "        results = []\n",
        "        for idx, distance in zip(indices[0], distances[0]):\n",
        "            if idx < len(self.chunks):\n",
        "                chunk = self.chunks[idx]\n",
        "                results.append({\n",
        "                    'content': chunk.page_content,\n",
        "                    'metadata': chunk.metadata,\n",
        "                    'similarity_score': float(1 / (1 + distance)),\n",
        "                    'chunk_id': int(idx)\n",
        "                })\n",
        "\n",
        "        return results\n",
        "\n",
        "    # ============= GENERATION METHODS =============\n",
        "\n",
        "    def generate_response(self, query: str, context_chunks: List[Dict]) -> str:\n",
        "        \"\"\"Generate response using local model\"\"\"\n",
        "\n",
        "        if not context_chunks:\n",
        "            return \"No relevant information found in the documents.\"\n",
        "\n",
        "        # Prepare context\n",
        "        context_parts = []\n",
        "        for chunk in context_chunks[:3]:\n",
        "            source = chunk['metadata'].get('source', 'Unknown')\n",
        "            page = chunk['metadata'].get('page', 'Unknown')\n",
        "\n",
        "            content = chunk['content'].replace('\\n', ' ').strip()\n",
        "            content = ' '.join(content.split())\n",
        "\n",
        "            context_parts.append(f\"[Source: {source}, Page {page}]\\n{content}\")\n",
        "\n",
        "        context = \"\\n\\n\".join(context_parts)\n",
        "\n",
        "        # Create prompt\n",
        "        prompt = f\"\"\"Answer the question based ONLY on the following context. If the context doesn't contain the answer, say \"I don't have enough information.\"\n",
        "\n",
        "Context:\n",
        "{context}\n",
        "\n",
        "Question: {query}\n",
        "\n",
        "Answer (be specific and concise):\"\"\"\n",
        "\n",
        "        # Truncate if too long\n",
        "        max_context_length = 1500\n",
        "        if len(prompt) > max_context_length:\n",
        "            prompt = prompt[:max_context_length] + \"...\"\n",
        "\n",
        "        # Try to generate response\n",
        "        if self.load_generation_model():\n",
        "            try:\n",
        "                if \"t5\" in self.config.generation_model.lower():\n",
        "                    inputs = self.tokenizer(prompt, return_tensors=\"pt\", max_length=512, truncation=True)\n",
        "\n",
        "                    with torch.no_grad():\n",
        "                        outputs = self.model.generate(\n",
        "                            inputs.input_ids,\n",
        "                            max_new_tokens=self.config.max_new_tokens,\n",
        "                            temperature=self.config.temperature,\n",
        "                            do_sample=self.config.do_sample,\n",
        "                            num_beams=1\n",
        "                        )\n",
        "\n",
        "                    response = self.tokenizer.decode(outputs[0], skip_special_tokens=True)\n",
        "\n",
        "                    if not response or len(response) < 10:\n",
        "                        return self._extract_answer_from_context(query, context_chunks)\n",
        "\n",
        "                    return response\n",
        "\n",
        "                elif self.generation_pipeline:\n",
        "                    response = self.generation_pipeline(prompt)[0]['generated_text']\n",
        "\n",
        "                    if not response or len(response) < 10:\n",
        "                        return self._extract_answer_from_context(query, context_chunks)\n",
        "\n",
        "                    return response\n",
        "\n",
        "            except Exception as e:\n",
        "                print(f\"Generation error: {e}\")\n",
        "                return self._extract_answer_from_context(query, context_chunks)\n",
        "        else:\n",
        "            return self._extract_answer_from_context(query, context_chunks)\n",
        "\n",
        "    def _extract_answer_from_context(self, query: str, context_chunks: List[Dict]) -> str:\n",
        "        \"\"\"Extract specific answers from context\"\"\"\n",
        "\n",
        "        if not context_chunks:\n",
        "            return \"No relevant information found in the documents.\"\n",
        "\n",
        "        query_lower = query.lower()\n",
        "\n",
        "        # Titanic survival question\n",
        "        if \"survive\" in query_lower or \"survived\" in query_lower:\n",
        "            for chunk in context_chunks:\n",
        "                content = chunk['content']\n",
        "\n",
        "                if \"Survived:\" in content:\n",
        "                    survived_match = re.search(r\"Survived:\\s*(\\d)\", content)\n",
        "                    name_match = re.search(r\"Name:\\s*([^,]+)\", content)\n",
        "\n",
        "                    if survived_match and name_match:\n",
        "                        survived = survived_match.group(1)\n",
        "                        name = name_match.group(1)\n",
        "\n",
        "                        if survived == \"1\":\n",
        "                            return f\"{name} survived the Titanic disaster.\"\n",
        "                        else:\n",
        "                            return f\"{name} did not survive the Titanic disaster.\"\n",
        "\n",
        "        # Check for GDP queries\n",
        "        if \"gdp of united states\" in query_lower or \"us gdp\" in query_lower or \"united states gdp\" in query_lower:\n",
        "            for chunk in context_chunks:\n",
        "                content = chunk['content']\n",
        "                if \"United States\" in content:\n",
        "                    lines = content.split('\\n')\n",
        "                    for line in lines:\n",
        "                        if \"United States\" in line:\n",
        "                            gdp_match = re.search(r'\\$(\\d+\\.?\\d*)', line)\n",
        "                            if gdp_match:\n",
        "                                return f\"The GDP of the United States is ${gdp_match.group(1)} trillion according to the document (Page {chunk['metadata'].get('page', 'Unknown')}).\"\n",
        "\n",
        "        # Generic response\n",
        "        response = f\"Based on the document, here's what I found about '{query}':\\n\\n\"\n",
        "\n",
        "        for i, chunk in enumerate(context_chunks[:2], 1):\n",
        "            source = chunk['metadata'].get('source', 'Unknown')\n",
        "            page = chunk['metadata'].get('page', 'Unknown')\n",
        "            content = chunk['content'].replace('\\n', ' ').strip()\n",
        "            content = ' '.join(content.split())\n",
        "            if len(content) > 300:\n",
        "                content = content[:300] + \"...\"\n",
        "\n",
        "            response += f\"{i}. From {source}, Page {page}:\\n\"\n",
        "            response += f\"   {content}\\n\\n\"\n",
        "\n",
        "        return response\n",
        "\n",
        "    # ============= QUERY METHOD =============\n",
        "\n",
        "    def query(self, query: str, k: int = None) -> Dict:\n",
        "        \"\"\"Complete query pipeline\"\"\"\n",
        "        print(f\"\\n{'='*60}\")\n",
        "        print(f\"QUERY: {query}\")\n",
        "        print(f\"{'='*60}\")\n",
        "\n",
        "        print(\"\\n🔍 Searching for relevant information...\")\n",
        "        results = self.search(query, k)\n",
        "\n",
        "        if not results:\n",
        "            print(\"No relevant information found.\")\n",
        "            return {\n",
        "                'query': query,\n",
        "                'response': \"No relevant information found in the documents.\",\n",
        "                'sources': []\n",
        "            }\n",
        "\n",
        "        print(f\"\\nFound {len(results)} relevant sections:\")\n",
        "        for i, result in enumerate(results, 1):\n",
        "            source = result['metadata'].get('source', 'Unknown')\n",
        "            page = result['metadata'].get('page', 'Unknown')\n",
        "            print(f\"\\n  {i}. [{source}, Page {page}] (Relevance: {result['similarity_score']:.2f})\")\n",
        "            preview = result['content'].replace('\\n', ' ').strip()\n",
        "            print(f\"     Preview: {preview[:100]}...\")\n",
        "\n",
        "        print(f\"\\n🤔 Generating answer...\")\n",
        "        response = self.generate_response(query, results)\n",
        "\n",
        "        return {\n",
        "            'query': query,\n",
        "            'response': response,\n",
        "            'sources': [\n",
        "                {\n",
        "                    'source': r['metadata']['source'],\n",
        "                    'page': r['metadata']['page'],\n",
        "                    'content': r['content'][:200] + \"...\",\n",
        "                    'relevance': r['similarity_score']\n",
        "                }\n",
        "                for r in results\n",
        "            ]\n",
        "        }\n",
        "\n",
